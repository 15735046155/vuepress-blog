# 一、普通函数

# this指向

  • this指向跟函数定义的位置没有关
  • this指向跟函数调用的位置、函数调用的方式有关

# this指向规则

# 默认绑定（独立函数调用）

指向window / undefined / {} --- 非严格模式 / 严格模式 / node环境（默认全局对象是个空对象）

// 方案一
function foo() {
    console.log(this) // window
}
foo()
    
// 闭包中的this
function fn(){
    function foo() {
        console.log(this) // window
    }
    return foo
}

var obj = { 
    name: 'hjj', 
    fn: fn
}

var bar = obj.fn()
bar()

# 隐式绑定: 指向某个对象

隐式绑定一般是通过某个对象的方法来调用

// 方案一
var obj = { 
    name: 'hjj', 
    fn: function fn(){
        console.log(this) // {name: 'hjj', fn: ƒ}
    }
}

 obj.fn()
 
// 方案二
function fn(){
    console.log(this) // {name: 'hjj', fn: ƒ}
}

var obj = { 
    name: 'hjj', 
    fn: fn
}

obj.fn()

# 显式绑定：指向现实绑定的对象

# call方法: 参数1:this的绑定对象，后序参数用逗号 , 隔开

function fn(num1, num2){
  this.num1 = num1
  this.num2 = num2
  console.log(this) // {name: 'hjj', age: 18, num1: 3, num2: 4}
}

var obj = { name: 'hjj', age: 18 }

fn.call(obj, 3, 4)

# apply方法：参数1:this的绑定对象，后序参数使用数组传递

function fn(num1, num2){
  this.num1 = num1
  this.num2 = num2
  console.log(this) // {name: 'hjj', age: 18, num1: 5, num2: 6}
}

var obj = { name: 'hjj', age: 18 }

fn.apply(obj, [5, 6])

# bind方法，返回一个新的函数

// bind返回的是一个新的函数
function fn(num1, num2){
    this.num1 = num1
    this.num2 = num2
    console.log(this)
}

  var obj = { name: 'hjj', age: 18 }

var foo = fn.bind(obj, 1,2)
var bar = fn

console.log(foo === fn) // false
console.log(fn === bar) // true

// bind使用
function fn(num1, num2){
    this.num1 = num1
    this.num2 = num2
    console.log(this) // {name: 'hjj', age: 18, num1: 1, num2: 2}
}

var obj = { name: 'hjj', age: 18 }

var foo = fn.bind(obj, 1, 2) // 返回的是一个函数，可以调用

foo(1, 2)

var p = new Person('hjj') // Person {name: 'hjj'} 'hjj'

# 内置函数绑定

  • settimeout: 内部函数是独立调用，所以this指向window(非箭头函数)

  • 元素onclick点击事件，或者是addEventListener指向元素本身

var box = document.getElementById("id")

box.onclick = function() {
    console.log(this) // box对应的元素
}
  • 数组的方法，高阶函数foreach,map...
var arr = ['hjj', "hsh"]

arr.forEach(function(item) {
    console.log(this) // Window {window: Window, self: Window, document: document, name: '', location: Location, …}
})

arr.forEach(function(item) {
    console.log(this) // String {'abc'}
}, "abc")

# this指向优先级

new > 显式绑定 > 隐式绑定 > 默认绑定

# 二、箭头函数

  • 箭头函本身没有this
  • 箭头函数不使用普通函数的规则
  • 箭头函数的this根据外层作用域来决定
// 箭头函数不绑定this.找上层作用域，他的上层作用域是第4行函数的作用域，此时第四行的作用域是obj
var obj = {
  data: [],
  getData: function() {
    setTimeout(() => {
        console.log(this) // {data: Array(0), getData: ƒ}
    }, 0);
  }
}
obj.getData();
// 更上一个例子的区别是，getData也是一个箭头函数，也没有this,继续向上找，直到最后是window
var obj = {
  data: [],
  getData: () => {
    setTimeout(() => {
        console.log(this) // window
    }, 0);
  }
}
obj.getData();

# 三、面试题

# 题一

var name = "window";
var person = {
  name: "person",
  sayName: function () {
    console.log(this.name);
  }
};
function sayName() {
  var sss = person.sayName;
  sss();
  person.sayName();
  (person.sayName)();
  (b = person.sayName)();
}
sayName();


// 答案
function sayName() {
  var sss = person.sayName;
  sss(); // window
  person.sayName(); // person
  (person.sayName)();  // person
  (b = person.sayName)();  // window
}

# 题二

var name = 'window'
var person1 = {
  name: 'person1',
  foo1: function () {
    console.log(this.name)
  },
  foo2: () => console.log(this.name),
  foo3: function () {
    return function () {
      console.log(this.name)
    }
  },
  foo4: function () {
    return () => {
      console.log(this.name)
    }
  }
}

var person2 = { name: 'person2' }

person1.foo1(); 
person1.foo1.call(person2); 

person1.foo2();
person1.foo2.call(person2);

person1.foo3()();
person1.foo3.call(person2)();
person1.foo3().call(person2);

person1.foo4()();
person1.foo4.call(person2)();
person1.foo4().call(person2);


// 答案
person1.foo1(); // person1
person1.foo1.call(person2); // person2

person1.foo2(); // window
person1.foo2.call(person2); //window

person1.foo3()(); // window
person1.foo3.call(person2)(); // window
person1.foo3().call(person2); // person2

person1.foo4()(); // person1
person1.foo4.call(person2)(); // person2
person1.foo4().call(person2); // person1
